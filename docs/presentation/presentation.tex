\begin{frame}{Задача ранжирования в рекомендательных системах}

Дано:
\begin{itemize}
  \item $\mathcal{U}$ — множество пользователей;
  \item $\mathcal{I}$ — множество объектов, в работе рассматриваются фильмы;
  \item история взаимодействий пользователя:
  $
    \mathcal{H}_u = \{(i, r_{ui}, t_{ui})\},
  $
  где $i$ — фильм, $t_{ui}$ — момент взаимодействия.
\end{itemize}


Найти:
Для пользователя $u$ нужно построить персональный список рекомендаций:
$$
  \operatorname{TopK}(u) = (i_1, i_2, \ldots, i_K),
$$
такой, что наиболее релевантные фильмы находятся выше в списке.


Задачи делятся по типам обратной связи:
\begin{itemize}
  \item \textbf{Implicit feedback:} известен только факт взаимодействия пользователя с фильмом.
  \item \textbf{Explicit feedback:} известна численная оценка пользователя.
  $
    r_{ui}\in\{0.5,1.0,\ldots,5.0\}
  $
\end{itemize}


\end{frame}

\begin{frame}{Цель работы}

\begin{itemize}
  \item собрать датасет и сгенерировать дополнительные признаки фильмов;
  \item сравнить базовые модели рекомендаций с графовыми архитектурами;
  \item исследовать способы использования рейтинга в графовых моделях;
  \item оценить качество моделей при обычном сценарии, а также для новых пользователей;
  \item сравнить время работы моделей.
\end{itemize}

\end{frame}

\begin{frame}{Метрики качества}


\textbf{HitRate@K}.



$$
\operatorname{HitRate@K}(u)
=
\mathbb{I}\{R_K(u)\cap G(u)\ne \varnothing\}.
$$

$$
\operatorname{HitRate@K}
=
\frac{1}{|\mathcal{U}_{eval}|}
\sum_{u\in\mathcal{U}_{eval}}
\operatorname{HitRate@K}(u).
$$

\textbf{NDCG@K}.

$$
\operatorname{NDCG@K}(u)
=
\frac{\operatorname{DCG@K}(u)}
{\operatorname{IDCG@K}(u)},
\qquad
\operatorname{DCG@K}(u)
=
\sum_{j=1}^{K}
\frac{2^{g_{u\pi_j}}-1}{\log_2(j+1)}.
$$


$$
\operatorname{IDCG@K}(u)
=
\sum_{j=1}^{K}
\frac{2^{g_{u\pi^{*}_j}}-1}{\log_2(j+1)},
$$

где $\pi^{*}$ — идеальный порядок фильмов по убыванию полезности.



\footnotesize
\begin{minipage}{0.96\linewidth}
$R_K(u)$ — первые $K$ рекомендаций для $u$; 
$G(u)$ — множество релевантных фильмов; 
$\pi_j$ — фильм на позиции $j$; 
$g_{ui}$ — полезность фильма $i$ для $u$.
\end{minipage}

\end{frame}

\begin{frame}{Функция потерь для ранжирования}

\begin{block}{BPR-loss}
$
  \mathcal{L}_{\mathrm{BPR}}
  =
  -\sum_{(u,i^+,i^-)}
  \log \sigma\big(s(u,i^+) - s(u,i^-)\big)
$, 

где u - пользователь, $i^+$ и $i^-$ положительный и отрицательный кандидаты.

\vspace{0.25em}

\footnotesize
BPR-loss напрямую оптимизирует относительный порядок объектов
в персональном ранжировании.
\end{block}

\begin{block}{Мера релевантности}
$
  s(u,i)=\langle \mathbf{z}_u, \mathbf{z}_i \rangle,
  \qquad
  \mathbf{z}_u=f_{\theta}(\mathcal{H}_u),
  \quad
  \mathbf{z}_i=g_{\phi}(\mathbf{x}_i).
$

\vspace{0.25em}

\footnotesize
$\mathbf{z}_u$ — представление пользователя по истории $\mathcal{H}_u$;
$\mathbf{z}_i$ — представление фильма по признакам $\mathbf{x}_i$;
$f_{\theta}$ и $g_{\phi}$ — сеть по пользователям и сеть по объектам.
\end{block}

\end{frame}

\begin{frame}{NGCF: графовая парадигма}

\small

\textbf{Граф пользователь-объект}
Взаимодействия пользователей и фильмов задают двудольный граф:

$$
\mathcal{G}=(\mathcal{U}\cup\mathcal{I},\mathcal{E}),
\qquad
(u,i)\in\mathcal{E}.
$$

\vspace{0.6em}

\textbf{Механизм сообщений.}

$$
\mathbf{e}^{(k+1)}_u =
\sigma\left(
\sum_{i\in\mathcal{N}(u)}
\frac{1}{\sqrt{d_u d_i}}
\left(
\mathbf{W}^{(k)}_1\mathbf{e}^{(k)}_i
+
\mathbf{W}^{(k)}_2
\big(\mathbf{e}^{(k)}_i \odot \mathbf{e}^{(k)}_u\big)
\right)
\right)
$$

\vspace{0.6em}

\textbf{Итоговое представление.}
Представления всех слоёв конкатенируются:

$$
\mathbf{z}_u =
\mathbf{e}^{(0)}_u \Vert \mathbf{e}^{(1)}_u \Vert \ldots \Vert \mathbf{e}^{(L)}_u,
\qquad
s(u,i)=\langle\mathbf{z}_u,\mathbf{z}_i\rangle
$$

\vspace{0.5em}

\footnotesize
$\mathcal{N}(u)$ — соседи пользователя;
$d_u,d_i$ — степени вершин;
$\mathbf{W}^{(k)}_1,\mathbf{W}^{(k)}_2$ — обучаемые матрицы слоя;
$\odot$ — покомпонентное произведение.

\end{frame}

\begin{frame}{GCMC}

\begin{figure}
    \centering
    \includegraphics[width=0.8\linewidth]{figures/GCMC.png}
    \caption{\scriptsize
    Архитектура GCMC.
    }
    \label{fig:gcmc_slide}
\end{figure}

\vspace{-0.3em}

\textbf{Кодировщик:}
$
  \mathbf{h}_u =
  \sigma\left(
  \operatorname{accum}_{q\in\mathcal{R}}
  \sum_{i\in\mathcal{N}_q(u)}
  \frac{1}{c_{ui}} \mathbf{W}_q\mathbf{x}_i
  \right),
  \qquad
  \mathbf{z}_u=\sigma(\mathbf{W}\mathbf{h}_u)
$

\vspace{0.45em}

\textbf{Декодировщик:}
$
  p(\hat r_{ui}=q)=
  \frac{\exp(\mathbf{z}_u^\top \mathbf{Q}_q\mathbf{z}_i)}
  {\sum_{s\in\mathcal{R}}\exp(\mathbf{z}_u^\top \mathbf{Q}_s\mathbf{z}_i)}
$

\vspace{0.45em}

\textbf{Функция потерь:}
$
  \mathcal{L}_{\mathrm{CE}}
  =
  -\sum_{(u,i)\in\Omega}
  \sum_{q\in\mathcal{R}}
  \mathbb{I}[r_{ui}=q]\log p(\hat r_{ui}=q)
$

\end{frame}

\begin{frame}{Эксперимент: функции рейтингового веса}
\small

\textbf{Встраивание рейтинга в механизм сообщений NGCF}

$$
\mathbf{e}^{(k+1)}_u =
\sigma\left(
\sum_{i\in\mathcal{N}(u)}
\frac{\textcolor{red}{w(r_{ui})}}{\sqrt{d_u d_i}}
\left(
\mathbf{W}^{(k)}_1\mathbf{e}^{(k)}_i
+
\mathbf{W}^{(k)}_2
\big(\mathbf{e}^{(k)}_i \odot \mathbf{e}^{(k)}_u\big)
\right)
\right)
$$

Линейная:
$$
w_{\mathrm{lin}}(r)
=
\frac{r-r_{\min}}{r_{\max}-r_{\min}}.
$$


\textbf{Предлагаемые функции рейтингового веса}

Обучаемая экспоненциальная:
$$
w_{\mathrm{exp}}(r)
=
\frac{\exp(\beta r)-\exp(\beta r_{\min})}
{\exp(\beta r_{\max})-\exp(\beta r_{\min})},
\qquad
\beta>0.
$$
Обучаемая softplus-функция:

$$
w_{\mathrm{spline}}(r)
=
\operatorname{softplus}
\left(
\sum_{m=1}^{M} a_m B_m(r)
\right)
\qquad
\operatorname{softplus}(x)=\log(1+\exp(x))
$$

\footnotesize
$B_m(r)$ — базисные сплайн-функции;
$a_m$ — обучаемые параметры.

\end{frame}
